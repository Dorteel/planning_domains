(define (problem hanoi-problem)
(:domain hanoi)
(:objects peg1 peg2 peg3 green_disk red_disk)
(:init
(on green_disk peg1)
(on red_disk peg2)
(clear green_disk)
(clear red_disk)
(peg peg1)
(peg peg2)
(peg peg3)
(smaller red_disk green_disk)
)
(:goal (and (on red_disk peg3) (on green_disk red_disk))))

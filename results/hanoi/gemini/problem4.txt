(define (problem hanoi-problem)
(:domain hanoi)
(:objects peg1 peg2 peg3 orange_disk red_disk blue_disk pink_disk)
(:init
(on pink_disk peg1)
(on blue_disk pink_disk)
(on red_disk blue_disk)
(on orange_disk red_disk)
(clear pink_disk)
(peg peg1)
(peg peg2)
(peg peg3)
(smaller blue_disk pink_disk)
(smaller red_disk blue_disk)
(smaller orange_disk red_disk)
)
(:goal (and (on orange_disk peg3) (on red_disk orange_disk) (on blue_disk red_disk) (on pink_disk blue_disk))))

(define (problem hanoi)
(:domain hanoi)
(:objects peg1 peg2 peg3 red_disk blue_disk green_disk)
(:init
(on green_disk peg1)
(on blue_disk green_disk)
(on red_disk blue_disk)
(clear green_disk)
(peg peg1)
(peg peg2)
(peg peg3)
(smaller blue_disk green_disk)
(smaller red_disk blue_disk)
)
(:goal (and (on red_disk peg3) (on blue_disk red_disk) (on green_disk blue_disk))))
